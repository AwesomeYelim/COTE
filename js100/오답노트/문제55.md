# 문제55. 하노이 탑


<mark style="background: pink">괄호문자열 ->'{','}','[',']','(',')'</mark><br>





#### 미리 생각해 봐야할 것

- 6가지 케이스를 살펴봄

> 1. (())
> 2. (((()()())))
> 3. ()()()
> 4. )))((( -> 순서가 맞아야함/ ()->(o)  )(->(x)
> 5. ())
> 6. ())(



- for 문내 if 문을 선언한다

```js
const route = [];

function hanoi(num, start, end, temp){
  // 원판이 한개 일때는 바로 옮기면 됩니다.
  if (num === 1){
    route.push([start, end]);
    return NaN;
  }

  // 원반이 n-1개를 경유기둥으로 옮기고
  hanoi()
  // 가장 큰 원반은 목표기둥으로
  route.push()
  // 경유기둥과 시작기둥을 바꿉니다.
  hanoi()
}

hanoi(3, 'A', 'B', 'C');
console.log(route);
console.log(route.length);
  
```
